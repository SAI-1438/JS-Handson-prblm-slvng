Q1.Find the Smaller Angle

var Minimal_Angle = (h, m) => 
      {
          h=h%12;
        let hour = (h*360)/12 + (m*360)/(12*60);
        let min=(m*360)/60;
        let angle= Math.abs(hour-min);
        
        if(angle>180)
        return 360-angle;
        else
        return angle;
      };

Q2.Check whether the year is Leap year or not.

var Check_Leap = (year) => 
        {
         if(year%400===0){
           return "Leap Year";
         }
         else if(year%100===0){
           return "Non Leap Year";
         }
         else if(year%4===0){
           return "Leap Year";
         }
         return "Non Leap Year";
          
        
        };

Q3.Perfect Number Check.

var Perfect_Check = (N) => 
        {
          let ans=0;
          for(let i=1;i<=N/2;i++){
            if(N%i===0){
              ans+=i;
            }
          }
          if(ans===N && ans!==0){
            return "YES";
          }
          else{
            return "NO";
          }
        };

Q4.Reverse a Number

var Reverse_Number = (N) => 
        {
         let ans=0;
         while(N>0){
           ryt=N%10;
